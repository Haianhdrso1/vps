-- Hàm phát hiện admin trong server
function Security:DetectAdmins()
    local admins = {}
    for _, player in pairs(Players:GetPlayers()) do
        -- Kiểm tra tên
        if table.find(AdminList, player.Name) then
            table.insert(admins, player)
        end
        
        -- Kiểm tra badges admin
        local success, badges = pcall(function()
            return MarketplaceService:UserOwnsGamePassAsync(player.UserId, 0)
        end)
        
        -- Kiểm tra group rank (giả sử Roblox Admin group ID = 1200769)
        local success2, rank = pcall(function()
            return player:GetRankInGroup(1200769)
        end)
        if success2 and rank and rank >= 100 then
            table.insert(admins, player)
        end
    end
    return admins
end

-- Hàm chống quét (Anti-Detection)
function Security:AntiDetection()
    -- Ẩn script khỏi detection
    local function hideScript()
        -- Đổi tên script thành random
        local randomName = "System_" .. HttpService:GenerateGUID(false)
        
        -- Tạo fake environment
        local fakeEnv = {}
        setmetatable(fakeEnv, {
            __index = getfenv(),
            __newindex = function() end
        })
    end
    
    -- Anti memory scan
    local function antiMemoryScan()
        local garbage = {}
        for i = 1, 1000 do
            table.insert(garbage, HttpService:GenerateGUID())
        end
    end
    
    -- Anti RemoteSpy
    local function antiRemoteSpy()
        for _, v in pairs(ReplicatedStorage:GetDescendants()) do
            if v:IsA("RemoteEvent") or v:IsA("RemoteFunction") then
                -- Fake name to confuse detection
                pcall(function()
                    v.Name = HttpService:GenerateGUID(false)
                end)
            end
        end
    end
    
    -- Anti Log
    local function disableLogs()
        -- Disable developer console
        pcall(function()
            setfflag("AbuseReportScreenshot", "False")
            setfflag("AbuseReportScreenshotPercentage", "0")
        end)
    end
    
    hideScript()
    antiMemoryScan()
    antiRemoteSpy()
    disableLogs()
end

-- Hàm tự động thoát khi phát hiện admin
function Security:AutoLeaveOnAdmin()
    local admins = Security:DetectAdmins()
    if #admins > 0 then
        Security.AdminInServer = true
        Security.AdminUsername = admins[1].Name
        
        -- Thông báo
        Library:Notify("⚠️ NGUY HIỂM!", "Admin " .. admins[1].Name .. " trong server!", 5, "🚨")
        
        -- Delay và thoát
        task.wait(2)
        Library:Notify("Thoát Server", "Đang rời server sau 3 giây...", 3, "🏃")
        task.wait(3)
        
        -- Teleport đến server khác
        local servers = {}
        local success, pages = pcall(function()
            return HttpService:JSONDecode(
                game:HttpGet("https://games.roblox.com/v1/games/" .. game.PlaceId .. "/servers/Public?sortOrder=Asc&limit=100")
            )
        end)
        
        if success and pages and pages.data then
            for _, server in ipairs(pages.data) do
                if server.playing < server.maxPlayers then
                    TeleportService:TeleportToPlaceInstance(game.PlaceId, server.id, Player)
                    return
                end
            end
        end
        
        -- Nếu không tìm thấy server khác, rejoin
        TeleportService:Teleport(game.PlaceId, Player)
    end
end

        ButtonFrame.Text = name
        ButtonFrame.Font = Enum.Font.Gotham
        ButtonFrame.TextSize = IsMobile and 14 or 13
        ButtonFrame.BorderSizePixel = 0
        ButtonFrame.Parent = parent
        
        ButtonFrame.MouseButton1Click:Connect(callback)
        ButtonFrame.TouchTap:Connect(callback) -- Mobile touch support
        
        return ButtonFrame
    end
    
    return Window
end
