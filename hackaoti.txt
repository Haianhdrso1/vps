local function deepBlockRemotes(container)
    for _, child in pairs(container:GetChildren()) do
        if child:IsA("RemoteEvent") or child:IsA("RemoteFunction") then
            local name = child.Name
            for _, pattern in ipairs(blockedRemotePatterns) do
                if name:find(pattern) then
                    local oldFire = child.FireServer
                    child.FireServer = function(...) return nil end
                    child.OnServerEvent = function() return nil end
                    if child:IsA("RemoteFunction") then
                        child.OnServerInvoke = function() return nil end
                    end
                    break
                end
            end
        elseif #child:GetChildren() > 0 then
            deepBlockRemotes(child)
        end
    end
end

local playerScanner = {
    knownPlayers = {},
    scanInterval = 5,
    alerts = {},
    scanHistory = {}
}

local function getDetailedPlayerInfo(plr)
    local info = {
        name = plr.Name,
        displayName = plr.DisplayName,
        userId = plr.UserId,
        team = plr.Team and plr.Team.Name or "None",
        character = plr.Character ~= nil,
        health = 0,
        position = Vector3.new(0,0,0),
        hasAdmin = false,
        ping = 0,
        isExploiter = false,
        detectedScripts = {},
        distanceFromUs = 0
    }
    if plr.Character and plr.Character:FindFirstChild("Humanoid") then
        info.health = plr.Character.Humanoid.Health
        if plr.Character:FindFirstChild("HumanoidRootPart") then
            info.position = plr.Character.HumanoidRootPart.Position
        end
    end
    if LP.Character and LP.Character:FindFirstChild("HumanoidRootPart") then
        info.distanceFromUs = (info.position - LP.Character.HumanoidRootPart.Position).Magnitude
    end
    local groups = plr:GetRoleInGroup(12345678)
    if groups and groups ~= "" then
        info.hasAdmin = true
    end
    info.ping = math.random(20, 200)
    return info
end

local function fullPlayerScan()
    local results = {}
    local allPlayers = Players:GetPlayers()
    for _, plr in ipairs(allPlayers) do
        if plr ~= LP then
            local info = getDetailedPlayerInfo(plr)
            table.insert(results, info)
            table.insert(playerScanner.scanHistory, {
                timestamp = os.time(),
                playerName = info.name,
                distance = info.distanceFromUs,
                health = info.health
            })
            if #playerScanner.scanHistory > 1000 then
                table.remove(playerScanner.scanHistory, 1)
            end
        end
    end
    return results
end

spawn(function()
    while true do
        local scanResult = fullPlayerScan()
        playerScanner.knownPlayers = scanResult
        for _, info in ipairs(scanResult) do
            if info.distanceFromUs < 150 and info.health > 1000 then
                table.insert(playerScanner.alerts, {
                    type = "SuspiciousHealth",
                    player = info.name,
                    health = info.health,
                    time = os.time()
                })
            end
            if info.hasAdmin then
                if info.distanceFromUs < 100 then
                    _G.stealthMode = true
                    _G.noclipEnabled = false
                    _G.speedHack = false
                    task.wait(5)
                    if info.distanceFromUs > 150 then
                        _G.stealthMode = false
                    end
                end
            end
        end
        wait(playerScanner.scanInterval)
    end
end)

            billboard.Parent = titan
            
            local label = Instance.new("TextLabel")
            label.Size = UDim2.new(1, 0, 1, 0)
            label.BackgroundTransparency = 1
            label.Text = "🩸 ТИТАН | HP: " .. (titan:FindFirstChild("Humanoid") and math.floor(titan.Humanoid.Health) or "???")

            label.TextScaled = true
            label.Parent = billboard
            
            espObjects[titan] = billboard
        end
    else
        if espObjects[titan] then
            espObjects[titan]:Destroy()
            espObjects[titan] = nil
        end
    end
end

        if _G["move_Полет (режим)"] then

                if flying then
                    char.Humanoid.PlatformStand = true
                    flyBV = Instance.new("BodyVelocity")
                    flyBV.MaxForce = Vector3.new(4000, 4000, 4000)
                    flyBV.Parent = char.HumanoidRootPart
                    spawn(function()
                        while flying and char and char.Parent do
                            local move = Vector3.new(0,0,0)
                            if UserInputService:IsKeyDown(Enum.KeyCode.W) then move = move + char.HumanoidRootPart.CFrame.LookVector end
                            if UserInputService:IsKeyDown(Enum.KeyCode.S) then move = move - char.HumanoidRootPart.CFrame.LookVector end
                            if UserInputService:IsKeyDown(Enum.KeyCode.A) then move = move - char.HumanoidRootPart.CFrame.RightVector end
                            if UserInputService:IsKeyDown(Enum.KeyCode.D) then move = move + char.HumanoidRootPart.CFrame.RightVector end
                            if UserInputService:IsKeyDown(Enum.KeyCode.Space) then move = move + Vector3.new(0,1,0) end
                            if UserInputService:IsKeyDown(Enum.KeyCode.LeftControl) then move = move - Vector3.new(0,1,0) end
                            flyBV.Velocity = move * 80
                            task.wait()
                        end
                        if flyBV then flyBV:Destroy() end
                        char.Humanoid.PlatformStand = false
                    end)
                else
                    if flyBV then flyBV:Destroy() end
                    char.Humanoid.PlatformStand = false
                end
            end
        end
    end
end)
